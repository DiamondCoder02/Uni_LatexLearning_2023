\subsection{Loops}

\newcounter{money}
\newcommand{\MoneyCounter}[1]{%
\setcounter{money}{#1}%
\whiledo{\value{money}>0}{%
\$\decrcounter{money}%
}
}
Furry artist salary: \MoneyCounter{10}\\
My salary: \MoneyCounter{1}

\newcounter{n}
\newcounter{ne}
\newcounter{nee}
\newcounter{neee}
\newcommand{\fib}[1]{%
\ifthenelse{#1=0}{\setcounter{neee}{0}}{
\ifthenelse{#1=1}{\setcounter{neee}{1}}{
\setcounter{ne}{0}
\setcounter{nee}{1}
\setcounter{neee}{1}
\setcounter{n}{#1}
\whiledo{\value{n}>1}{
\addtocounter{neee}{\value{ne}}
\setcounter{ne}{\value{nee}}
\setcounter{nee}{\value{neee}}
\addtocounter{n}{-1}
}
}
}\theneee
}
\fib{4}
